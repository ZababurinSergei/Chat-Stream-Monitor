Компонент для работы с вкладками.
Структура вкладки/таба:

```typescript
interface Tab {
    /**
     * Отображаемое название таба в заголовке вкладки.
     */
    title: string;

    /**
     * Уникальный идентификатор представления (view), связанного с табом.
     * Используется для поиска, восстановления и сравнения вкладок.
     */
    viewId: string;

    /**
     * Имя роута, к которому привязан таб (из конфигурации роутера).
     */
    routeName: string;

    /**
     * Опциаонально!
     * Полный путь роута, который открывается при активации таба
     * Хранится для возможности восстановления навигации.
     */
    routePath?: string;

    /**
     * Флаг, обозначающий, что данные, связанные с табом, были изменены
     * и отличаются от исходного/сохранённого состояния.
     */
    isModified: boolean;

    /**
     * Пользовательские данные, связанные с содержимым таба
     * (например, форма, выбранные фильтры и т.п.).
     */
    userData: any;

    /**
     * Флаг, определяющий, можно ли закрыть таб (показывать кнопку "Закрыть").
     */
    isClosable: boolean;

    /**
     * Опциаонально!
     * Параметры роута или дополнительные данные,
     * по которым можно однозначно идентифицировать вкладку.
     */
    params?: any;

    /**
     * Опциаонально!
     * Дополнительное имя расширения/сущности,
     * которое может переопределять стандартный заголовок таба.
     */
    extensionName?: string | null;
}
```
